% \cventry{date}{title}{institution}{location}{grade}{description}
\newcommand{\cventry}[6]{
  \noindent
  \begin{tabular*}{\textwidth}{@{\extracolsep{\fill}} l r @{}}
    \textbf{#2} & \textit{#4} \\                       % title | location
    \textit{#3}, #1 & \ifblank{#5}{}{#5} \\            % institution, date | grade
    \ifblank{#6}{}{\multicolumn{2}{@{}p{\textwidth}@{}}{\small #6} \\}  % description
  \end{tabular*}
  \par\addvspace{\cvitemsep}
}

% \cvitem{label}{content}
\newcommand{\cvitem}[2]{%
  \par\addvspace{\cvitemsep}%
  \noindent
  \makebox[\cvlabelwidth][l]{#1}\hspace{0.8em}%
  \parbox[t]{\dimexpr\textwidth-\cvlabelwidth-0.8em\relax}{#2}%
  \par
}

% \cvdoubleitem{label1}{text1}{label2}{text2}
\newcommand{\cvdoubleitem}[4]{%
  \par\addvspace{\cvitemsep}%
  \noindent
  \makebox[\cvlabelwidth][l]{#1}\hspace{0.8em}%
  \parbox[t]{0.3\textwidth}{#2}%
  \hfill
  \makebox[\cvlabelwidth][l]{#3}\hspace{0.8em}%
  \parbox[t]{0.3\textwidth}{#4}%
  \par
}
